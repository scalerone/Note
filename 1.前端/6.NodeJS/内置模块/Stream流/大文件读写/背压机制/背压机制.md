## 代码示例

  - 示例1

    ```javascript
    const fs = require('fs');

    const rs = fs.createReadStream('./test.txt', {
      highWaterMark: 4
    });
    const ws = fs.createWriteStream('./test2.txt', {
      highWaterMark: 1
    });

    let flag = true;

    rs.on('data', (chunk) => {
      flag = ws.write(chunk, () => {
        console.log(111);
      });
      if(!flag) {
        rs.pause();
      }

      console.log(flag);
    });

    ws.on('drain', () => {
      // 继续执行
      rs.resume();
    })
    ```

  - 示例2

    ```javascript
    const fs = require('fs');

    });

    rs.pipe(ws);
    ```
