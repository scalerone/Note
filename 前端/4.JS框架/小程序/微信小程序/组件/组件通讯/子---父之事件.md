# 子-->父之事件

## 使用

*   子组件

    ```javascript
    <view bindtap="two" data-author-name="12"></view>
    ```

    ```javascript
    methods: {
      two(e) {
        const a = e.currentTarget.dataset.authorName;
        this.triggerEvent('towLeve', a);  // 传递参数
      }
    }
    ```

*   父组件

    ```javascript
    // 注册对应事件
    <view bind:towLeve="towLeve"></view>
    ```

    ```javascript
    towLeve(e) {
      const datas = e.detail;  // 获取参数
    }
    ```
