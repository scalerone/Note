# 对象遍历

## 倾向于对象遍历而不是Switch语句

+ 示例

  ```js
  function test(color) {
    // 使用条件语句来寻找对应颜色的水果
    switch (color) {
      case 'red':
        return ['apple', 'strawberry'];
      case 'yellow':
        return ['banana', 'pineapple'];
      case 'purple':
        return ['grape', 'plum'];
      default:
        return [];
    }
  }

  // test results
  test(null); // []
  test('yellow'); // ['banana', 'pineapple']
  ```

+ 对象遍历

  ```js
  const fruitColor = {
    red: ['apple', 'strawberry'],
    yellow: ['banana', 'pineapple'],
    purple: ['grape', 'plum'],
    'default': ['other','Index'],
  };

  function test(color) {
    return fruitColor[color] || fruitColor['default'];
  }
  ```

## 使用 Map

+ 使用 `Map`

  ```js
  const actions = new Map([
    [1, ['processing','IndexPage']],
    [2, ['fail','FailPage']],
    [3, ['fail','FailPage']],
    [4, ['success','SuccessPage']],
    [5, ['cancel','CancelPage']],
    ['default', ['other','Index']]
  ])

  const onButtonClick = (status)=>{
    let action = actions.get(status) || actions.get('default')
  }

  onButtonClick(2); // ["fail", "FailPage"]
  ```

## 不仅需要判断状态，还需要判断用户的身份

+ 常规

  ```js
  const onButtonClick = (status,identity)=>{
    if(identity == 'guest'){
      if(status == 1){
        //do sth
      }else if(status == 2){
        //do sth
      }else if(status == 3){
        //do sth
      }else if(status == 4){
        //do sth
      }else if(status == 5){
        //do sth
      }else {
        //do sth
      }
    }else if(identity == 'master') {
      if(status == 1){
        //do sth
      }else if(status == 2){
        //do sth
      }else if(status == 3){
        //do sth
      }else if(status == 4){
        //do sth
      }else if(status == 5){
        //do sth
      }else {
        //do sth
      }
    }
  }
  ```

+ 解决方案

  ```js
  const actions = new Map([
    ['guest_1', ()=>{/*do sth*/}],
    ['guest_2', ()=>{/*do sth*/}],
    ['guest_3', ()=>{/*do sth*/}],
    ['guest_4', ()=>{/*do sth*/}],
    ['guest_5', ()=>{/*do sth*/}],
    ['master_1', ()=>{/*do sth*/}],
    ['master_2', ()=>{/*do sth*/}],
    ['master_3', ()=>{/*do sth*/}],
    ['master_4', ()=>{/*do sth*/}],
    ['master_5', ()=>{/*do sth*/}],
    ['default', ()=>{/*do sth*/}],
  ])

  const onButtonClick = (identity,status)=>{
    let action = actions.get(`${identity}_${status}`) || actions.get('default')
    action.call(this)
  }
  ```
