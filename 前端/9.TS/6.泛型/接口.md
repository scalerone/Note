# 接口

## 接口

  - 直接在名称后 `<泛型名称>`

    ```typescript
    interface Cb<T> {
      (n:T, i:number):boolean
    }

    function filter<T>(arr:T[], cb:Cb<T>):T[] {
      const newArr:T[] = []
      arr.forEach((n, i:number) => {
        if(cb(n, i)) {
          newArr.push(n)
        }
      });
      return newArr
    }

    const arr = filter<number>([54,6,5,46], (item) => {
      return (item > 10);
    })
    console.log(arr) // [ 54, 46 ]
    ```
